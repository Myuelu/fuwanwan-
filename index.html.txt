    <style>
        body {
            margin: 0;
            overflow: hidden;
            background-color: #f0f0f0; /* 背景色（好みに合わせて変えてね） */
        }
        canvas {
            display: block;
            width: 100vw;
            height: 100vh;
        }
    </style>
    <!-- 必要なライブラリをネット上から読み込む -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/pixi.js/5.3.12/pixi.min.js"></script>
    <script src="https://cdn.jsdelivr.net/gh/arthur970304/pixi-live2d-display/dist/index.min.js"></script>
</head>
<body>
    <canvas id="canvas"></canvas>

    <script>
        const app = new PIXI.Application({
            view: document.getElementById('canvas'),
            autoStart: true,
            resizeTo: window,
            backgroundAlpha: 0 /* 背景を透明にする */
        });

        // ここに自分の「model3.json」へのパスを書く
        // HTMLと同じフォルダにあるなら、ファイル名だけでOK！
        const modelUrl = 'fuwanwan.model3.json'; 

        PIXI.live2d.Live2DModel.from(modelUrl).then(model => {
            app.stage.addChild(model);

            // モデルのサイズと位置の自動調整
            const scaleX = app.screen.width / model.width;
            const scaleY = app.screen.height / model.height;
            const scale = Math.min(scaleX, scaleY) * 0.8; // 画面の8割の大きさに調整
            model.scale.set(scale);
            
            // 画面の中央に配置
            model.x = (app.screen.width - model.width) / 2;
            model.y = (app.screen.height - model.height) / 2;

            // マウス追従を有効化（これで勝手にカーソルを追いかける！）
            model.trackPointer();
        });
    </script>
</body>
